SELECT CTL.LOAN_APP_SEQ,
       CTL.CNIC_NUM,
       CTL.RMKS,
       TO_CHAR (CTL.EFF_START_DT, 'dd-MM-yyyy') AS EFF_START_DT,
       TO_CHAR (CTL.TAG_TO_DT, 'dd-MM-yyyy') AS TAG_TO_DT,
       TO_CHAR (CTL.TAG_FROM_DT, 'dd-MM-yyyy') AS TAG_FROM_DT,
       CTL.LAST_UPD_BY,
       TO_CHAR (CTL.LAST_UPD_DT, 'dd-MM-yyyy') AS LAST_UPD_DT
  FROM MW_CLNT_TAG_LIST CTL
 WHERE CTL.TAGS_SEQ = 6 AND :whereTagType
 ORDER BY CTL.LAST_UPD_DT DESC
