SELECT VAL.REF_CD_VAL_CODE, VAL.REF_CD_VAL_DESC
  FROM ADC.ADC_REF_CD_VAL  VAL
       JOIN ADC.ADC_REF_CD_GRP GRP
           ON     GRP.REF_CD_GRP_CODE = VAL.REF_CD_GRP_CODE
              AND VAL.CRNT_REC_FLG = GRP.CRNT_REC_FLG
              AND VAL.REF_CD_VAL_CODE NOT IN ('0002', '0003', '0008')
 WHERE GRP.CRNT_REC_FLG = 1 AND GRP.REF_CD_GRP_CODE = '0001'
