SELECT SLS.SANC_SEQ,
       SLS.NATIONAL_ID,
       SLS.CNIC_NUM,
       SLS.FIRST_NM,
       SLS.LAST_NM,
       SLS.FATHER_NM,
       SLS.CNTRY,
       SLS.PRVNCE,
       SLS.DSTRCT,
       TRUNC (SLS.DOB),
       SLS.REF_NO,
       SLS.SANC_TYPE
  FROM MW_SANC_LIST SLS
 WHERE     SLS.CRNT_REC_FLG = :P_CRNT_REC_FLG
       AND SLS.IS_MTCH_FOUND = :P_IS_MTCH_FOUND
       AND UPPER (SLS.CNTRY) = 'PAKISTAN'
       AND UPPER (SLS.SANC_TYPE) LIKE '%' || :P_SANC_TYP || '%'
